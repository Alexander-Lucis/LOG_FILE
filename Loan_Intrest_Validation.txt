Q4.Write an ASP.Net program to design a form having loan amount, interest rate and duration fields. Calculate the simple interest and perform necessary validation i.e. Ensures data has been entered for each field. Checking for non-numeric value. Assume suitable web-form controls and perform necessary validation.

BE (Default.aspx.cs):
using System;
using System.Collections.Generic;
using System.Linq;
using System.Web;
using System.Web.UI;
using System.Web.UI.WebControls;

namespace LoanCalculatorApp
{
    public partial class WebForm1 : System.Web.UI.Page
    {
        protected void btnCalculate_Click(object sender, EventArgs e)
        {
            if (string.IsNullOrWhiteSpace(txtLoanAmount.Text) ||
                string.IsNullOrWhiteSpace(txtInterestRate.Text) ||
                string.IsNullOrWhiteSpace(txtDuration.Text))
            {
                lblResult.Text = "Please enter values for all fields.";
                return;
            }

            double loanAmount, interestRate, duration;
            if (!double.TryParse(txtLoanAmount.Text, out loanAmount) ||
                !double.TryParse(txtInterestRate.Text, out interestRate) ||
                !double.TryParse(txtDuration.Text, out duration))
            {
                lblResult.Text = "Please enter valid numeric values for all fields.";
                return;
            }

            double simpleInterest = (loanAmount * interestRate * duration) / 100;
            lblResult.Text = $"Simple Interest: {simpleInterest:C}";
        }
    }
}
